import { Form } from './form';

## React Component for creating forms

An extensible react form component.

### Component usage
```js
const headings = [
    {
        uid: 'foo',
        label: 'Foo',
    }
]
const customFieldComponents = {
    /* Map of field type to component */
}
<Form
    headings={headings}
    onSubmit={({ formEditData, formData }) => {}}
    onChange={() => {}}
    customFieldComponents={customFieldComponents}
/>
```

### Using props to customize the text

Modify the text to see it change live:
```js live
<Form
    headings={[
        {
            uid: 'foo',
            label: 'Foo',
            type: 'text',
        }
    ]}
    onSubmit={({ formEditData, formData }) => {}}
    onChange={() => {}}
    customFieldComponents={{
        'text': () => 'placeholder text',
    }}
/>
```
